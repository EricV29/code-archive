.ORIG x4000

; Ejemplo: Suma de dos numeros
;----------------------------------------
LD R1, NUMU 
LD R2, NUMD

INICIA
       JSR SUM
       JSR IMP
       

IMP
       OUT
       HALT
       
SUM
       ADD R5, R7, #0
       ADD R0, R1, R2
       ADD R3, R3, R0
       ADD R7, R5, #0
       RET


NUMU .FILL x3
NUMD .FILL x7
.END
