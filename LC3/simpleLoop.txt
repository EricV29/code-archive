.ORIG x8400
;----------------------------------------
; Ejemplo: Simple Loop
;----------------------------------------
LD R1, BMAX ;CARGA EL VALOR DE LOOPMAX EN R1
LD R2, NEG ;CARGA EL VALOR DE NEG_ONE EN R1
LEA R0, MENSAJE ;CARGA DIRECCION DE LOOP_MSG EN R0

LOOP_INICIO
PUTS ;IMPRIMIME EL MENSAJE DEL BUCLE
ADD R1, R2, R1 ; DISMINUYE EL CONTADOR DEL BUCLE R1
;NOTA: ADD R1, #-1 TAMBIEN FUNCIONARIA ARRIBA
BRp LOOP_INICIO ;SI R1 TODAVIA EL POSITIVO, REALIZAR EL BUCLE DE NUEVO
LOOP_END ;LA ETIQUETA NO HACE NADA, PERO DE LO CONTRARIO A LA CONDICION PASADA SE TERMINA

LEA R0, DONE_MSG ;CARGAR LA DIRECCION DEL MENSAJE
PUTS ;IMPRIME EL MENSAJE HECHO

HALT ;DETENER 

BMAX .FILL x000B ;HEXADECIMAL PARA 10
NEG .FILL #-1 ;DECIAL PARA -1
MENSAJE .STRINGZ "Holaaa\n  Como estas?\n"
DONE_MSG .STRINGZ "Adiosss!\n"
.END
