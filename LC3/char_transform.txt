.ORIG x3000
LD R3, MT
LD R1, LM
LD R2, NEG
LD R5, SUM

LOOP_INICIO
LD R4, CE
IN   
ADD R4, R0, R4
ADD R4, R4, R5 
ADD R4, R4, R3
ADD R0, R4, #0
OUT
LD R0, NEWLINE
OUT 
ADD R1, R2, R1
BRp LOOP_INICIO
LOOP_END

HALT

NEWLINE .FILL x000D
CE .FILL x0
NEG .FILL #-1
SUM .FILL x30
LM .FILL x4
MT .FILL x3
.END
